38  /***
39   * 
40   * starts and intializes all the settings the SocialSite Application
41   * 
42   * @author Ananth
43   */
44  public class SocialSiteApplication extends WebApplication
45  {
46  	private static final long serialVersionUID = 1L;
47

49  	{
50  		return (SocialSiteApplication)Application.get();
51  	}
52  
53  	/***
54  	 * Constructor
55  	 */
56  	public SocialSiteApplication()
57  	{
58  	}
59  
60  	@Override
61  	public String getConfigurationType()
62  	{
63  		return "DEVELOPMENT";
64  		// return "DEPLOYMENT";
65  	}
66  
67  	/***
68  	 * @see org.apache.wicket.Application#getHomePage()
69  	 */
70  	@Override
71  	public Class<LoginPage> getHomePage()
72  	{
73  		return LoginPage.class;
74  	}
75  
76  	/*
77  	 * overide the init method to initialize the springDAo
78  	 */
79  	@Override
80  	protected void init()
81  	{
82  
83  		super.init();
84  
85  		getDebugSettings().setDevelopmentUtilitiesEnabled(true);
86  
87  		// add spring listener to the application
88  		addComponentInstantiationListener(new SpringComponentInjector(this));
89  
90  
91  		// intializes the security listeners
92  		initSecurity();
93  
94  		// enables wicket autolinking
95  		getMarkupSettings().setAutomaticLinking(true);
96  
97  		// init all the image resources
98  		initImageResources();
99  
100 
101 		// mounts the activation page
102 		mount(new QueryStringUrlCodingStrategy("activate", ActivationPage.class));
103 
104 		// logs all the request to the server
105 		// Application.get().getRequestLoggerSettings().setRequestLoggerEnabled(true);
106 	}
107 
108 	/***
109 	 * registers all the image resources
110 	 */
111 	private void initImageResources()
112 	{
113 		// user image
114 		getSharedResources().add(ImageType.USER.name(), new UserImageResource());
115 		// university image
116 		getSharedResources().add(ImageType.UNIVERSITY.name(), new UniversityImageResource());
117 		// course image
118 		getSharedResources().add(ImageType.COURSE.name(), new CourseImageResource());
119 
120 		// notes
121 		getSharedResources().add("note", new CourseNoteResource());
122 
123 	}
124 
125 	/***
126 	 * initializes all the security settings
127 	 * 
128 	 */
129 	private void initSecurity()
130 	{
131 		// set the authorization Strategy
132 		getSecuritySettings().setAuthorizationStrategy(
133 				new RoleAuthorizationStrategy(new UserRolesAuthorizer()));
134 
135 	}
136 
137 	/***
138 	 * 
139 	 * defining the custom session
140 	 * 
141 	 * @return custom session object
142 	 */
143 	@Override
144 	public Session newSession(final Request request, final Response response)
145 	{
146 		return new SocialSiteSession(request);
147 	}
