57  public class SignUpPage extends WebPage
58  {
59  	/*** Model object for username */
60  	private String userName;
61  	/*** Model object for password */
62  	private String password;
63  	/*** Model object for password */
64  	private String rePassword;
65  
66  	/*** Model object for email */
67  	private String email;
68  
69  	/*** Model object for university name */
70  	private String universityName;
71  
72  	/*** Model for the userType **/
73  	private UserType userType;
74  
75  	/*** Feedback panel */
76  	private final FeedbackPanel feedback;
77  
78  	/*** Spring Dao to handle profile object */
79  	@SpringBean(name = "profileDao")
80  	private ProfileDao profileDao;
81  
82  	/*** Spring Dao to handle user object */
83  	@SpringBean(name = "userDao")
84  	private UserDao<User> userDao;
85  
86  	public SignUpPage()
87  	{
88  
89  		// sign up form
90  		final Form<Object> form = new Form<Object>("signupform");
91  		add(form);
92  
93  		final TextField<String> username = new RequiredTextField<String>("username",
94  				new PropertyModel<String>(this, "userName"));
95  		username.add(StringValidator.maximumLength(16));
96  
97  		form.add(username);
98  
99  		final TextField<String> emailTextField = new RequiredTextField<String>("email",
100 				new PropertyModel<String>(this, "email"));
101 		emailTextField.add(EmailAddressValidator.getInstance());
102 
103 		form.add(emailTextField);
104 
105 		final PasswordTextField passwordTextField = new PasswordTextField("password",
106 				new PropertyModel<String>(this, "password"));
107 		passwordTextField.setRequired(true);
108 		passwordTextField.add(StringValidator.lengthBetween(6, 16));
109 
110 		form.add(passwordTextField);
111 
112 		final PasswordTextField rePasswordTextField = new PasswordTextField("re-password",
113 				new PropertyModel<String>(this, "rePassword"));
114 		rePasswordTextField.setRequired(true);
115 		rePasswordTextField.add(StringValidator.lengthBetween(6, 16));
116 
117 		form.add(rePasswordTextField);
118 
119 		final RadioGroup<UserType> userGroup = new RadioGroup<UserType>("usertype",
120 				new PropertyModel<UserType>(this, "userType"));
121 
122 		userGroup.add(new Radio<UserType>("student", new Model<UserType>(UserType.STUDENT)));
123 		userGroup.add(new Radio<UserType>("staff", new Model<UserType>(UserType.STAFF)));
124 		userGroup.add(new Radio<UserType>("admin", new Model<UserType>(UserType.ADMIN)));
125 		form.add(userGroup);
126 
127 		form.add(new TextField<String>("university", new PropertyModel<String>(this,
128 				"universityName")));
129 
130 		SubmitLink signUp;
131 		form.add(signUp = new SubmitLink("signup")
132 		{
133 
134 			/***
135 			 * 
136 			 */
137 			private static final long serialVersionUID = 1L;
138 
139 			@Override
140 			public void onSubmit()
141 			{
142 
143 				if (!password.equals(rePassword))
144 				{
145 					error("passwords should be same");
146 					return;
147 				}
148 
149 				// creates a user
150 				try
151 				{
152 					User user = null;
153 
154 					switch (userType)
155 					{
156 						case STUDENT :
157 							user = new Student(userName, password);
158 							break;
159 						case STAFF :
160 							user = new Staff(userName, password);
161 							break;
162 						case ADMIN :
163 							if (universityName != null && !universityName.equals(""))
164 							{
165 								user = new Admin(userName, password);
166 								userDao.save(user);
167 								new UniversityActivator((Admin)user, universityName).create();
168 								// final InfoMsg message = new InfoMsg();
169 								// message.getUsers().add(user);
170 								// message.setMessage("Your request for adding new University is being processed");
171 								// messageDao.save(message);
172 							}
173 							else
174 							{
175 								error("University Name required");
176 								return;
177 							}
178 
179 							break;
180 						default :
181 							error("Unknown error occured ");
182 							return;
183 					}
184 
185 					userDao.save(user);
186 
187 					final Profile p = new Profile();
188 					p.setUser(user);
189 					user.setProfile(p);
190 					p.setEmail(email);
191 					// set the default image for the profile
192 					new DefaultImage().forUser(p);
193 					profileDao.save(p);
194 
195 					final SocialSiteSession session = SocialSiteSession.get();
196 					session
197 							.setSessionUser(new SessionUser(user.getId(), SocialSiteRoles.ownerRole));
198 					session.setUserId(user.getId());
199 					setResponsePage(new HomePage());
200 
201 				}
202 				catch (final ConstraintViolationException ex)
203 				{
204 					// updates the feedback panel
205 					error("username already exists");
206 					// target.addComponent(feedback);
207 					return;
208 				}
209 			}
210 		});
211 		// submit the form on return key
212 		form.setDefaultButton(signUp);
213 
214 		// feedback panel
215 		feedback = new FeedbackPanel("feedback");
216 		feedback.setOutputMarkupId(true);
217 		add(feedback);
218 	}
