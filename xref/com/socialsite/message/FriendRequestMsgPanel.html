43   * @author Ananth
44   */
45  @AuthorizeAction(action = Action.RENDER, roles = "OWNER")
46  public class FriendRequestMsgPanel extends Panel
47  {
48  
49  	/***
50  	 * 
51  	 */
52  	private static final long serialVersionUID = 1L;
53  
54  	/*** spring Dao to handle user object */
55  	@SpringBean(name = "userDao")
56  	private UserDao<User> userDao;
57  
58  	@SpringBean(name = "messageDao")
59  	private MessageDao<Message> messageDao;
60  
61  	public FriendRequestMsgPanel(final String id, final FriendRequestMsg friendRequestMsg,
62  			final WebMarkupContainer container)
63  	{
64  		super(id);
65  
66  		UserLink<User> user;
67  		final User sender = friendRequestMsg.getSender();
68  		// user image
69  		UserLink<User> userImageLink;
70  		final Model<User> senderModel = new Model<User>(sender);
71  		add(userImageLink = new UserLink<User>("imagelink", senderModel));
72  		userImageLink.add(new ImagePanel("userthumb", sender.getId(), ImageType.USER, sender
73  				.getLastModified(), true));
74  		add(user = new UserLink<User>("user", new Model<User>(friendRequestMsg.getSender())));
75  		user.add(new Label("name", friendRequestMsg.getSender().getUserName()));
76  		add(new Label("message", friendRequestMsg.getMessage()));
77  		add(new Link<FriendRequestMsg>("yes", new Model<FriendRequestMsg>(friendRequestMsg))
78  		{
79  			/***
80  		 * 
81  		 */
82  			private static final long serialVersionUID = 1L;
83  
84  			@Override
85  			public void onClick()
86  			{
87  
88  				FriendRequestMsg friendRequestMsg = getModelObject();
89  
90  				friendRequestMsg = (FriendRequestMsg)messageDao.load(friendRequestMsg.getId());
91  
92  				// add him as the friend
93  				final User user = friendRequestMsg.getSender();
94  
95  				user.addFriend(friendRequestMsg.getUsers().iterator().next());
96  				// save the user
97  				userDao.save(user);
98  
99  				// remove the friend request
100 				messageDao.delete(friendRequestMsg);
101 
102 			}
103 		});
104 
105 		add(new AjaxLink<FriendRequestMsg>("no", new Model<FriendRequestMsg>(friendRequestMsg))
106 		{
107 
108 			/***
109 			 * 
110 			 */
111 			private static final long serialVersionUID = 1L;
112 
113 			@Override
114 			public void onClick(final AjaxRequestTarget target)
115 			{
116 				// remove the friend request
117 				messageDao.delete(getModelObject());
118 
119 				// update the list
120 				target.addComponent(container);
121 			}
122 		});
123 
124 		add(new Label("time", relativeTime(friendRequestMsg.getTime())));
125 
126 	}
127
