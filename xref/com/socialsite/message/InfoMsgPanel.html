46  /***
47   * @author Ananth
48   */
49  public class InfoMsgPanel extends BasePanel
50  {
51  	/***
52  	 * 
53  	 */
54  	private static final long serialVersionUID = 1L;
55  
56  	/*** model for reply text */
57  	private String text;
58  
59  	/*** feedback panel */
60  	FeedbackPanel feedback;
61  	/*** spring dao to handle message object */
62  	@SpringBean(name = "messageDao")
63  	private MessageDao<Message> messageDao;
64  
65  	/***
66  	 * constructor
67  	 * 
68  	 * @param id
69  	 *            id
70  	 * @param infoMsg
71  	 *            message
72  	 */
73  	public InfoMsgPanel(final String id, final IModel<InfoMsg> infoMsgModel,
74  			final MarkupContainer dependent)
75  	{
76  		super(id, infoMsgModel);
77  		final InfoMsg infoMsg = infoMsgModel.getObject();
78  		final User sender = infoMsg.getSender();
79  
80  		// user image
81  		UserLink<User> userImageLink;
82  		final Model<User> senderModel = new Model<User>(sender);
83  		add(userImageLink = new UserLink<User>("imagelink", senderModel));
84  		userImageLink.add(new ImagePanel("userthumb", sender.getId(), ImageType.USER, sender
85  				.getLastModified(), true));
86  		Link<User> name;
87  		add(name = new UserLink<User>("home", senderModel));
88  		name.add(new Label("username", sender.getUserName()));
89  
90  
91  		add(new Label("date", DateUtils.relativeTime(infoMsg.getTime())));
92  
93  		// message
94  		add(new Label("message", infoMsg.getMessage()).setEscapeModelStrings(false));
95  
96  		// delete link
97  		add(new DeleteMsgLink<InfoMsg>("delete", infoMsgModel, dependent, this, infoMsg.getSender()
98  				.getId()));
99  
100 		// reply form
101 		final Form<InfoMsg> form = new Form<InfoMsg>("form", infoMsgModel);
102 		add(form);
103 		form.add(new RichEditor("richeditor", new PropertyModel<String>(this, "text")));
104 		form.add(new AjaxSubmitLink("send")
105 		{
106 
107 			/***
108 			 * 
109 			 */
110 			private static final long serialVersionUID = 1L;
111 
112 			@Override
113 			public boolean isVisible()
114 			{
115 				// don't allow the user to reply himself
116 				if (infoMsg.getSender().getId() == getSessionUserId())
117 				{
118 					return false;
119 				}
120 				return hasRole(SocialSiteRoles.OWNER);
121 			}
122 
123 			@Override
124 			protected void onError(final AjaxRequestTarget target, final Form<?> form)
125 			{
126 				super.onError(target, form);
127 				// show feedback messages
128 				target.addComponent(feedback);
129 			}
130 
131 			@Override
132 			protected void onSubmit(final AjaxRequestTarget target, final Form<?> form)
133 			{
134 				try
135 				{
136 					final InfoMsg msg = (InfoMsg)form.getModelObject();
137 					final InfoMsg replyMsg = new InfoMsg();
138 					replyMsg.setMessage(text);
139 					replyMsg.setSender(getSessionUser());
140 					replyMsg.setTime(new Date());
141 					replyMsg.addUser(msg.getSender());
142 					messageDao.save(replyMsg);
143 				}
144 				catch (final Exception ex)
145 				{
146 					target
147 							.appendJavascript("SocialSite.Message.show('sorry we can't send your message. Try again later');");
148 				}
149 
150 				// show a message
151 				target
152 						.appendJavascript("SocialSite.Message.show('Your reply message has been sent successfully ');");
153 
154 				// slideup the reply panel
155 				final String id = InfoMsgPanel.this.getMarkupId();
156 				target.appendJavascript(" $('#" + id + " .slideText').trigger('click'); ");
157 
158 
159 				// TODO show the success message
160 				// either use a global success or failure message handler
161 				// or show the messge near this message
162 
163 				// clears the old feedback message
164 				target.addComponent(feedback);
165 			}
166 		});
167 
168 		// feedback for reply
169 		add(feedback = new FeedbackPanel("feedback"));
170 		feedback.setOutputMarkupId(true);
171 		setOutputMarkupId(true);
172 	}
173 }
