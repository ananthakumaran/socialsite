33  /***
34   * Abstract DAO implementation for all the Domain class
35   * 
36   * @author Ananth
37   */
38  public abstract class AbstractDaoImpl<T> implements AbstractDao<T>
39  {
40  
41  	/*** class of the domain */
42  	protected final Class<T> domainClass;
43  	/*** Hibernate sessionFactory to access the database */
44  	private SessionFactory sessionFactory;
45  
46  	/***
47  	 * constructor
48  	 * 
49  	 * @param domainClass
50  	 *            domainClass
51  	 */
52  	public AbstractDaoImpl(final Class<T> domainClass)
53  	{
54  		this.domainClass = domainClass;
55  	}
56  
57  	/***
58  	 * helper method to find the count of item that match the filter
59  	 * 
60  	 * @param filter
61  	 *            filter text
62  	 * @param clazz
63  	 *            class of the item
64  	 * @param field
65  	 *            field that will be matched with the criteria
66  	 * @return
67  	 */
68  	@SuppressWarnings("unchecked")
69  	protected int count(String filter, final Class clazz, final String field)
70  	{
71  		// Avoids NPE
72  		filter = filter == null ? "" : filter;
73  
74  		final Criteria criteria = getSession().createCriteria(clazz);
75  		criteria.add(Restrictions.ilike(field, filter, MatchMode.ANYWHERE));
76  		criteria.setProjection(Projections.rowCount());
77  		return (Integer)criteria.uniqueResult();
78  	}
79  
80  	/***
81  	 * find the number of unique rows in the given domainclass (table)
82  	 * 
83  	 * @return no of unique rows in the given domainclass (table)
84  	 */
85  	public int countAll()
86  	{
87  		final Criteria criteria = getSession().createCriteria(domainClass);
88  		criteria.setProjection(Projections.rowCount());
89  		return (Integer)criteria.uniqueResult();
90  	}
91  
92  	/***
93  	 * removes the given domainclass object from the Database
94  	 * 
95  	 * @param object
96  	 *            object to be deleted
97  	 */
98  	public void delete(final T object)
99  	{
100 		getSession().delete(object);
101 	}
102 
103 	/***
104 	 * a helper method to find the list of item matching the filter
105 	 * 
106 	 * @param filter
107 	 *            filter text
108 	 * @param first
109 	 *            first index of the resul
110 	 * @param count
111 	 *            count
112 	 * @param sortParam
113 	 *            sorting is done based on this property
114 	 * @param clazz
115 	 *            class of the item
116 	 * @param field
117 	 *            field that will be matched with the criteria
118 	 * @return list of item matching the filter text
119 	 */
120 	@SuppressWarnings("unchecked")
121 	protected List find(String filter, final int first, final int count, final SortParam sortParam,
122 			final Class clazz, final String field)
123 	{
124 		// avoids the NPE
125 		filter = filter == null ? "" : filter;
126 
127 		// order
128 		final Order order = sortParam.isAscending() ? Order.asc(sortParam.getProperty()) : Order
129 				.desc(sortParam.getProperty());
130 		final Criteria criteria = getSession().createCriteria(clazz);
131 		criteria.setFirstResult(first);
132 		criteria.setMaxResults(count);
133 		criteria.add(Restrictions.ilike(field, filter, MatchMode.ANYWHERE));
134 		criteria.addOrder(order);
135 		return criteria.list();
136 	}

138 	/***
139 	 * find all the rows in the table and return it as a list
140 	 * 
141 	 * @return List contains all the rows in the given domainclass (table)
142 	 */
143 	@SuppressWarnings("unchecked")
144 	public List<T> findAll()
145 	{

147 		return criteria.list();
