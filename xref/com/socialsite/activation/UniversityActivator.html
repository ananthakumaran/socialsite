18  package com.socialsite.activation;
19  
20  import org.apache.wicket.injection.web.InjectorHolder;
21  import org.apache.wicket.spring.injection.annot.SpringBean;
22  
23  import com.socialsite.dao.ActivationDao;
24  import com.socialsite.email.Email;
25  import com.socialsite.email.EmailSender;
26  import com.socialsite.persistence.Activation;
27  import com.socialsite.persistence.Admin;
28  
29  /***
30   * 
31   * @author Ananth
32   * 
33   */
34  public class UniversityActivator
35  {
36  	private final Admin admin;
37  	private final String universityName;
38  	private long activationId;
39  
40  	@SpringBean(name = "activationDao")
41  	ActivationDao activationDao;
42  
43  	public UniversityActivator(final Admin admin, final String universityName)
44  	{
45  		this.admin = admin;
46  		this.universityName = universityName;
47  		InjectorHolder.getInjector().inject(this);
48  	}
49  
50  	public long addActivation()
51  	{
52  		final Activation activation = new Activation();
53  		activation.setAdmin(admin);
54  		activation.setUniversityName(universityName);
55  		activationDao.save(activation);
56  		return activation.getId();
57  	}
58  
59  	public void create()
60  	{
61  		activationId = addActivation();
62  		// send email to the admin
63  		sendEmail();
64  	}
65  
66  	public String getActivationUrl()
67  	{
68  		return "http://localhost:8081/activate?id=" + activationId + "&action=activate";
69  	}
70  
71  	public String getDeActivationUrl()
72  	{
73  		return "http://localhost:8081/activate?id=" + activationId + "&action=deactivate";
74  	}
75  
76  	public String getMessage()
77  	{
78  		final StringBuffer message = new StringBuffer();
79  		message.append("University Activation Request \n");
80  		message.append("UniversityName : ").append(universityName).append("\n");
81  		message.append("Admin Name :").append(admin.getUserName()).append("\n");
82  		// message.append("Admin email :").append(admin.getProfile().getEmail()).append("\n");
83  		message.append("To activate the university click the following url ").append(
84  				getActivationUrl()).append("\n");
85  		message.append("To Deactivate the university click the following url ").append(
86  				getDeActivationUrl()).append("\n");
87  		return message.toString();
88  	}
89  
90  	/***
91  	 * sends email to the admin
92  	 */
93  	public void sendEmail()
94  	{
95  		final Email email = new Email();
96  		// am i sending message to myself :)
97  		email.addReceivers("ananthakumaran@gmail.com");
98  		email.setSubject("University Activation Request");
99  		email.setMessage(getMessage());
100 		new EmailSender().send(email);
101 	}
102 }
