70  	/***
71  	 * fires a event with the collection of all the updated dom elements after
72  	 * the wicket ajax response. To subscribe the event call the
73  	 * <code>SocialSite.Ajax.registerPostAjax</code>. Your callback function
74  	 * will be called with a jQuery Wrapped set of all the update dom as the
75  	 * first argument.
76  	 * 
77  	 * NOTE: call this only once after all the components are added to the
78  	 * target
79  	 * 
80  	 * 
81  	 * @param target
82  	 *            ajax target
83  	 */
84  	public void firePostAjaxUpdateEvent(final AjaxRequestTarget target)
85  	{
86  		final StringBuffer script = new StringBuffer(" SocialSite.Ajax.handle([");
87  		for (final Component component : target.getComponents())
88  		{
89  			script.append("\"" + component.getMarkupId() + "\",");
90  		}
91  		script.append("])");
92  
93  		target.getHeaderResponse().renderOnDomReadyJavascript(script.toString());
94  	}
