19

21  import org.apache.wicket.markup.html.panel.Panel;
22  import org.apache.wicket.spring.injection.annot.SpringBean;
23  
24  import com.socialsite.SocialSiteSession;
25  import com.socialsite.course.NewCoursePanel;
26  import com.socialsite.dao.ProfileDao;
27  import com.socialsite.dao.UserDao;
28  import com.socialsite.friend.AddAsFriendPanel;
29  import com.socialsite.image.ImagePanel;
30  import com.socialsite.image.ImageService;
31  import com.socialsite.image.ImageType;
32  import com.socialsite.persistence.Profile;
33  import com.socialsite.persistence.User;
34  
35  /***
36   * @author Ananth
37   */
38  public class UserInfoPanel extends Panel
39  {
40  
41  	/***
42  	 * 
43  	 */
44  	private static final long serialVersionUID = 1L;
45  	/*** spring dao to handle user object */
46  	@SpringBean(name = "userDao")
47  	private UserDao<User> userDao;
48  
49  	/*** spring dao to handle profile object */
50  	@SpringBean(name = "profileDao")
51  	private ProfileDao profileDao;
52  
53  
54  	public UserInfoPanel(final String id)
55  	{
56  		super(id);
57  
58  		final User user = userDao.load(SocialSiteSession.get().getUserId());
59  
60  		add(new ImagePanel("userimage", user.getId(), ImageType.USER, user.getLastModified())
61  		{
62  
63  			/***
64  			 * 
65  			 */
66  			private static final long serialVersionUID = 1L;
67  
68  			@Override
69  			protected void saveImage(final byte[] imageData)
70  			{
71  				final ImageService imageService = new ImageService();
72  
73  				user.getProfile().changeThumb(
74  						imageService.resize(imageData, ImageService.THUMB_SIZE));
75  				user.getProfile().changeImage(
76  						imageService.resize(imageData, ImageService.IMAGE_SIZE));
77  				profileDao.save(user.getProfile());
78  				userDao.save(user);
79  
80  			}
81  		});
82  
83  		final Profile profile = user.getProfile();
84  
85  		add(new Label("username", user.getUserName()));
86  		String city = "";
87  		if (profile.getCurrentCity() != null)
88  		{
89  			city = profile.getCurrentCity().getValue();
90  		}
91  		add(new Label("city", city));
92  		add(new Label("status", profile.getRelationshipStatus()));
93  		add(new Label("aboutme", profile.getAboutMe()));
94  		add(new AddAsFriendPanel("addasfriend"));
95  		add(new NewCoursePanel("addnewcourse"));
96  	}
97  }
