34  	private class OnUploadedBehavior extends AbstractDefaultAjaxBehavior
35  	{
36  		/***
37  		 * 
38  		 */
39  		private static final long serialVersionUID = 1L;
40  
41  		public String getCallback()
42  		{
43  			return generateCallbackScript(
44  					"wicketAjaxGet('" + getCallbackUrl(false)
45  							+ "&amp;amp;newFileUrl=' + encodeURIComponent(newFileUrl)"
46  							+ " + '&amp;amp;clientFileName=' + encodeURIComponent(clientFileName)")
47  					.toString();
48  		}
49  
50  		@Override
51  		protected void respond(final AjaxRequestTarget target)
52  		{
53  			onUploadFinished(target, getRequest().getParameter("clientFileName"), getRequest()
54  					.getParameter("newFileUrl"));
55  		}
56  	}

59  	 * 
60  	 */
61  	private static final long serialVersionUID = 1L;

63  	private InlineFrame uploadIFrame = null;

65  	public UploadPanel(final String id)
66  	{
67  		super(id);
68  		addOnUploadedCallback();
69  		setOutputMarkupId(true);

72  	/***
73  	 * Hackie method allowing to add a javascript in the page defining the
74  	 * callback called by the innerIframe
75  	 * 
76  	 */
77  	private void addOnUploadedCallback()
78  	{
79  		final OnUploadedBehavior onUploadBehavior = new OnUploadedBehavior();
80  		add(onUploadBehavior);
81  		add(new WebComponent("onUploaded")
82  		{
83  			/***
84  			 * 
85  			 */
86  			private static final long serialVersionUID = 1L;
87  
88  			@Override
89  			protected void onComponentTagBody(final MarkupStream markupStream,
90  					final ComponentTag openTag)
91  			{
92  				// calling it through setTimeout we ensure that the callback is
93  				// called
94  				// in the proper execution context, that is the parent frame
95  				replaceComponentTagBody(markupStream, openTag, "function onUpload_"
96  						+ UploadPanel.this.getMarkupId()
97  						+ "(clientFileName, newFileUrl) {window.setTimeout(function() { "
98  						+ onUploadBehavior.getCallback() + " }, 0 )}");
99  			}
100 		});
101 	}

103 	/***
104 	 * Create the iframe containing the upload widget
105 	 * 
106 	 */
107 	private void addUploadIFrame()
108 	{
109 		final IPageLink iFrameLink = new IPageLink()
110 		{
111 			/***
112 			 * 
113 			 */
114 			private static final long serialVersionUID = 1L;
115 
116 			public Page getPage()
117 			{
118 				return new UploadIFrame()
119 				{
120 					@Override
121 					protected String getOnUploadedCallback()
122 					{
123 						return "onUpload_" + UploadPanel.this.getMarkupId();
124 					}
125 
126 					@Override
127 					protected String manageInputSream(final FileUpload upload)
128 					{
129 						return UploadPanel.this.onFileUploaded(upload);
130 					}
131 				};
132 			}
133 
134 			public Class<UploadIFrame> getPageIdentity()
135 			{
136 				return UploadIFrame.class;
137 			}
138 		};
139 		uploadIFrame = new InlineFrame("upload", getPage().getPageMap(), iFrameLink);
140 		add(uploadIFrame);
141 	}
142 
143 	@Override
144 	protected void onBeforeRender()
145 	{
146 		super.onBeforeRender();
147 		if (uploadIFrame == null)
148 		{
149 			// the iframe should be attached to a page to be able to get its
150 			// pagemap,
151 			// that's why i'm adding it in onBeforRender
152 			addUploadIFrame();
153 		}
154 	}
