108 	 * helper to create profile for users
109 	 * 
110 	 * @param users
111 	 *            users
112 	 */
113 	public void createProfiles(final User... users)
114 	{
115 		final DefaultImage image = new DefaultImage();
116 
117 		for (final User user : users)
118 		{
119 			final Profile profile = new Profile();
120 			profile.setUser(user);
121 			user.setProfile(profile);
122 			image.forUser(user.getProfile());
123 			profile.setCurrentCity(new PrivacyModel("sample city", Access.EVERYONE));
124 			profile.setAboutMe("sample about me");
125 			profile.setEmail("sample" + user.getId() + "@gmail.com");
126 			profile.setSex("Male");
127 			profile.setRelationshipStatus("single");
128 			profileDao.save(user.getProfile());
129 		}
130 	}
